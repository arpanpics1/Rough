CREATE TABLE my_table (
    id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY (
        START WITH 33000
        INCREMENT BY 1
        MINVALUE 1
        MAXVALUE 99999999
        CACHE 20
        NOORDER
        NOCYCLE
    ),
    name VARCHAR2(100),
    id_with_prefix AS (TO_NUMBER('85' || TO_CHAR(id))) VIRTUAL
);
